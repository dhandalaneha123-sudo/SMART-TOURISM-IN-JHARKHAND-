        function nextSlide() {
            currentSlide = (currentSlide + 1) % totalSlides;
            updateCarousel();
        }
        
        function previousSlide() {
            currentSlide = (currentSlide - 1 + totalSlides) % totalSlides;
            updateCarousel();
        }
        
        function goToSlide(slideIndex) {
            currentSlide = slideIndex;
            updateCarousel();

        function updateCarousel() {
            const slides = document.querySelectorAll('.carousel-slide');
            const dots = document.querySelectorAll('.carousel-dot');
            
            slides.forEach((slide, index) => {
                if (index === currentSlide) {
                    slide.classList.remove('opacity-0');
                    slide.classList.add('active');
                } else {
                    slide.classList.add('opacity-0');
                    slide.classList.remove('active');
                }
            });
            
            dots.forEach((dot, index) => {
                if (index === currentSlide) {
                    dot.classList.add('active', 'bg-opacity-100');
                    dot.classList.remove('bg-opacity-50');
                } else {
                    dot.classList.remove('active', 'bg-opacity-100');
                    dot.classList.add('bg-opacity-50');
                }
            });
        }

        function sendChatMessage() {
            const input = document.getElementById('chat-input');
            const message = input.value.trim();
            
            if (!message) return;
            
            const messagesContainer = document.getElementById('chat-messages');
            
            // Add user message
            messagesContainer.innerHTML += `
                <div class="text-right mb-3">
                    <div class="bg-jharkhand-green text-white rounded-lg p-3 inline-block max-w-xs">
                        ${message}
                    </div>
                </div>
            `;
            
            // Simulate bot response
            setTimeout(() => {
                const botResponse = generateBotResponse(message);
                messagesContainer.innerHTML += `
                    <div class="bg-gray-100 rounded-lg p-3 mb-3">
                        <p class="text-sm">${botResponse}</p>
                    </div>
                `;
                messagesContainer.scrollTop = messagesContainer.scrollHeight;
            }, 1000);
            
            input.value = '';
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }

        function generateBotResponse(message) {
            const responses = {
                'weather': 'The weather in Jharkhand is pleasant from October to March. Summer can be hot, and monsoon brings beautiful waterfalls!',
                'transport': 'You can reach Jharkhand by train, bus, or flight to Ranchi. Local transport includes buses, taxis, and auto-rickshaws.',
                'food': 'Try local delicacies like Dhuska, Litti Chokha, Pitha, and tribal cuisines. Many restaurants serve authentic Jharkhandi food.',
                'hotels': 'We have partnered homestays, forest lodges, and hotels. Would you like me to help you find accommodation?',
                'default': 'That\'s interesting! For specific information about destinations, weather, transport, or bookings, feel free to ask. I\'m here to help make your Jharkhand trip amazing! üåø'
            };
            
            const lowerMessage = message.toLowerCase();
            
            if (lowerMessage.includes('weather') || lowerMessage.includes('climate')) {
                return responses.weather;
            } else if (lowerMessage.includes('transport') || lowerMessage.includes('travel') || lowerMessage.includes('reach')) {
                return responses.transport;
            } else if (lowerMessage.includes('food') || lowerMessage.includes('eat') || lowerMessage.includes('restaurant')) {
                return responses.food;
            } else if (lowerMessage.includes('hotel') || lowerMessage.includes('stay') || lowerMessage.includes('accommodation')) {
                return responses.hotels;
            } else {
                return responses.default;
            }
        }

        function sendSOSAlert() {

                    const lon = position.coords.longitude;
                    showNotification(`üö® SOS Alert sent! Location: ${lat.toFixed(4)}, ${lon.toFixed(4)}. Emergency services notified.`);
                }, function() {
                    showNotification('üö® SOS Alert sent! Unable to get location, but emergency services have been notified.');
                });
            } else {
                showNotification('üö® SOS Alert sent! Emergency services have been notified.');
            }
        }

        // Countdown Timer
        function startCountdownTimer() {
            let hours = 23;
            let minutes = 45;
            let seconds = 30;
            
            setInterval(() => {
                seconds--;
                if (seconds < 0) {
                    seconds = 59;
                    minutes--;
                    if (minutes < 0) {
                        minutes = 59;
                        hours--;
                        if (hours < 0) {
                            hours = 23;
                            minutes = 59;
                            seconds = 59;
                        }
                    }
                }
                
                document.getElementById('hours').textContent = hours.toString().padStart(2, '0');
                document.getElementById('minutes').textContent = minutes.toString().padStart(2, '0');
                document.getElementById('seconds').textContent = seconds.toString().padStart(2, '0');
            }, 1000);
        }
